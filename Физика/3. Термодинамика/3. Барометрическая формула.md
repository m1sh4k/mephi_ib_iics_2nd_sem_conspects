$$
\begin{align*}
&\vec{F}_{н}+\vec{F}_{в}+dm\vec{g}=0 && \text{По II з.н.}\\
&oy: \quad F_{н}-F_{в}=dmg \\
&\hline \\
&\cancel{ ps }-(\cancel{ p }+dp)s-dmg=0 \\
&-dps=dmg \tag{1}\\
&\boxed{pdV=\frac{{dm}}{\mu}RT} \tag{2}&  & \text{По уравнению Клайпейрона-Менделеева} \\ \\
&\text{Найдем }dm \text{ из (2)}\text{, подставим в (1)}\\
&dm=\frac{psdy\mu}{RT} \tag{3}\\
&-dp\cancel{ s }=\ \frac{psdy\mu}{RT}\cdot g\quad\bigg|\div p \\
&\frac{dp}{p }=-\frac{\mu gdy}{RT} && \text{Возьмем неопределенный интеграл}\\
&\ln p=-\frac{\mu g}{RT}y+C \\
\end{align*}

$$

С учетом начальных значений

$$
\begin{align*}
&p(0)=p_{0}\implies \ln p_{0}=C \\
&\ln p-\ln p_{0}=- \frac{{\mu g}}{RT}y \\
&\ln \frac{p}{p_{0}}=-\frac{\mu g}{RT}y \\
&\boxed{p=p_{0}e^{\textstyle\frac{-\mu gy}{RT}}}
\end{align*}
$$
